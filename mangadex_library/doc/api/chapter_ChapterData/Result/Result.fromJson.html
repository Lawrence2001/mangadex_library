Result.fromJson(Map<String, dynamic> json) {
  result = json['result'] ?? '';
  if (json['data'] != null) {
    data = Data.fromJson(json['data']);
  }
  if (json['relationships'] != null) {
    relationships = <Relationships>[];
    json['relationships'].forEach((v) {
      relationships.add(Relationships.fromJson(v));
    });
  }
}
